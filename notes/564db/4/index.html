PROCEDURES and FUNCTIONS

delimiter $$
drop procedure if exists getStudent;
create procedure getStudent(IN studentName VARCHAR(255)) 
begin
select *
from STUDENT
where sname = studentName;
end $$ 
delimiter ;

delimiter $$
Drop procedure if exists updateStudentTable; 
create procedure updateStudentTable() 
begin
update student set age =22 where sname ='Edward Baker';
update student set major = 'Architecture' where sname = 'Charles Harris'; 
end $$
delimiter ;
