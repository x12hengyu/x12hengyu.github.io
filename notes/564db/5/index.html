SELECT a.symbol, AVG(a.close)
FROM assets a, commo c
Where c.symbol = a.symbol AND a.symbol IN (
SELECT DISTINCT symbol 
FROM assets 
WHERE date BETWEEN '2005-01-01' AND '2005-12-31')
GROUP BY a.symbol
ORDER BY AVG(a.close) ASC
LIMIT 5;
